**Fix 1 — Critical: Replace `_poll_and_download` and `_download_output` in `comfyui_provider.py`**

Remove `_download_output` method entirely.

Replace `_poll_and_download` with this implementation (see RESEARCH.md "Corrected `_poll_and_download` Pattern"):
```python
async def _poll_and_download(
    self, client: httpx.AsyncClient, prompt_id: str, output_node: str
) -> bytes:
    """Poll until completed, fetch history, then download output image bytes."""
    elapsed = 0.0
    while elapsed < POLL_TIMEOUT:
        await asyncio.sleep(POLL_INTERVAL)
        elapsed += POLL_INTERVAL

        resp = await client.get(
            f"{settings.comfyui_base_url}/api/job/{prompt_id}/status",
            headers=self._headers(),
        )
        resp.raise_for_status()
        data = resp.json()
        status = data.get("status")
        logger.debug(f"ComfyUI job {prompt_id} status: {status}")

        if status == "completed":
            # Status body is {"status": "completed"} only.
            # Fetch outputs from history endpoint (separate API call).
            return await self._fetch_history_and_download(client, prompt_id, output_node)
        if status in ("failed", "cancelled", "canceled"):
            raise RuntimeError(
                f"ComfyUI job {status}: {data.get('error') or data}"
            )

    raise TimeoutError(f"ComfyUI job timed out after {POLL_TIMEOUT}s")


async def _fetch_history_and_download(
    self, client: httpx.AsyncClient, prompt_id: str, output_node: str
) -> bytes:
    """Fetch /api/history_v2/{prompt_id} and download the image from output node."""
    # Retry loop: outputs may not be persisted immediately after status=completed
    # (GitHub issue #11540 — execution_success fires before output write completes)
    outputs = {}
    for attempt in range(3):
        history_resp = await client.get(
            f"{settings.comfyui_base_url}/api/history_v2/{prompt_id}",
            headers=self._headers(),
        )
        history_resp.raise_for_status()
        history = history_resp.json()
        outputs = history.get("outputs", {})
        if outputs:
            break
        logger.warning(
            f"ComfyUI history outputs empty (attempt {attempt + 1}/3), retrying..."
        )
        await asyncio.sleep(2.0)

    node_out = outputs.get(output_node, {})
    images = node_out.get("images", [])
    if not images:
        logger.error(
            f"No images in output node {output_node}. Full history: {history}"
        )
        raise RuntimeError(
            f"ComfyUI: no images found in output node '{output_node}'. "
            f"Keys available: {list(outputs.keys())}"
        )

    img_meta = images[0]
    filename = img_meta["filename"]
    subfolder = img_meta.get("subfolder", "")

    view_resp = await client.get(
        f"{settings.comfyui_base_url}/api/view",
        params={"filename": filename, "subfolder": subfolder, "type": "output"},
        headers=self._headers(),
        follow_redirects=True,
    )
    if view_resp.is_error:
        logger.error(
            f"ComfyUI view error {view_resp.status_code}: {view_resp.text}"
        )
    view_resp.raise_for_status()
    logger.info(f"Downloaded {len(view_resp.content)} bytes from ComfyUI output")
    return view_resp.content
```
